def calculate_technical_indicators(self, nav_series):
    """技术指标补充"""
    # 20日移动平均（判断趋势）
    ma20 = nav_series.rolling(window=20).mean()
    
    # 当前价格偏离均线的程度（超买超卖信号）
    deviation = (nav_series.iloc[-1] - ma20.iloc[-1]) / ma20.iloc[-1] * 100
    
    # 简单判断：偏离>5%可能超买，<-5%可能超卖
    signal = "超买" if deviation > 5 else "超卖" if deviation < -5 else "正常"
    
    return {
        '20日均线偏离': round(deviation, 2),
        '信号': signal
    }

# 新建 fund_backtest.py
def backtest_moving_average_strategy(nav_series, short_ma=5, long_ma=20):
    """
    双均线策略回测
    金叉买入，死叉卖出
    """
    signals = []
    position = False  # False空仓, True持仓
    
    ma_short = nav_series.rolling(short_ma).mean()
    ma_long = nav_series.rolling(long_ma).mean()
    
    for i in range(long_ma, len(nav_series)):
        if ma_short.iloc[i] > ma_long.iloc[i] and not position:
            signals.append(('买入', nav_series.index[i], nav_series.iloc[i]))
            position = True
        elif ma_short.iloc[i] < ma_long.iloc[i] and position:
            signals.append(('卖出', nav_series.index[i], nav_series.iloc[i]))
            position = False
    
    # 计算最终收益
    return signals
