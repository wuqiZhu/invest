import time
from functools import wraps

def retry_on_failure(max_retries=3, delay=1):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_retries):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_retries - 1:
                        raise
                    time.sleep(delay * (attempt + 1))
            return None
        return wrapper
    return decorator
```

def calculate_technical_indicators(self, nav_series):
    """技术指标补充"""
    # 20日移动平均（判断趋势）
    ma20 = nav_series.rolling(window=20).mean()
    
    # 当前价格偏离均线的程度（超买超卖信号）
    deviation = (nav_series.iloc[-1] - ma20.iloc[-1]) / ma20.iloc[-1] * 100
    
    # 简单判断：偏离>5%可能超买，<-5%可能超卖
    signal = "超买" if deviation > 5 else "超卖" if deviation < -5 else "正常"
    
    return {
        '20日均线': round(ma20.iloc[-1], 4),
        '20日均线偏离': round(deviation, 2),
        '信号': signal
    }
```

def calculate_moving_averages(self, nav_series):
    """计算多周期均线"""
    return {
        'MA5': nav_series.rolling(5).mean().iloc[-1],
        'MA10': nav_series.rolling(10).mean().iloc[-1],
        'MA20': nav_series.rolling(20).mean().iloc[-1],
        'MA60': nav_series.rolling(60).mean().iloc[-1],
    }

def calculate_rsi(self, nav_series, period=14):
    """计算RSI指标"""
    delta = nav_series.diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    return rsi.iloc[-1]

def calculate_bollinger_bands(self, nav_series, period=20):
    """计算布林带"""
    ma = nav_series.rolling(period).mean()
    std = nav_series.rolling(period).std()
    upper = ma + 2 * std
    lower = ma - 2 * std
    return {
        '上轨': upper.iloc[-1],
        '中轨': ma.iloc[-1],
        '下轨': lower.iloc[-1]
    }
```

import requests

class FundNotifier:
    def send_wechat(self, content, webhook_url=None):
        """通过企业微信机器人发送通知（免费）"""
        if webhook_url is None:
            webhook_url = self.config.get('alert', {}).get('wechat_webhook')
        
        if not webhook_url:
            print("未配置微信webhook")
            return False
        
        data = {
            "msgtype": "text",
            "text": {"content": content}
        }
        
        try:
            response = requests.post(webhook_url, json=data, timeout=10)
            return response.status_code == 200

import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart

def send_email(self, subject, body, to_email=None):
    """通过QQ邮箱发送（免费）"""
    if to_email is None:
        to_email = self.config.get('alert', {}).get('email')
    
    smtp_server = "smtp.qq.com"
    smtp_port = 587
    sender_email = self.config.get('alert', {}).get('sender_email')
    sender_password = self.config.get('alert', {}).get('sender_password')
    
    msg = MIMEMultipart()
    msg['From'] = sender_email
    msg['To'] = to_email
    msg['Subject'] = subject
    msg.attach(MIMEText(body, 'plain', 'utf-8'))
    
    try:
        server = smtplib.SMTP(smtp_server, smtp_port)
        server.starttls()
        server.login(sender_email, sender_password)
        server.send_message(msg)
        server.quit()
        return True
    except Exception as e:
        print(f"邮件通知失败: {e}")
        return False
```

def backtest_moving_average_strategy(nav_series, short_ma=5, long_ma=20):
    """
    双均线策略回测
    金叉买入，死叉卖出
    """
    signals = []
    position = False  # False空仓, True持仓
    
    ma_short = nav_series.rolling(short_ma).mean()
    ma_long = nav_series.rolling(long_ma).mean()
    
    for i in range(long_ma, len(nav_series)):
        if ma_short.iloc[i] > ma_long.iloc[i] and not position:
            signals.append(('买入', nav_series.index[i], nav_series.iloc[i]))
            position = True
        elif ma_short.iloc[i] < ma_long.iloc[i] and position:
            signals.append(('卖出', nav_series.index[i], nav_series.iloc[i]))
            position = False
    
    return signals
```

def calculate_backtest_stats(signals, nav_series):
    """计算回测统计指标"""
    if not signals:
        return {}
    
    # 计算总收益
    initial_capital = 10000
    shares = 0
    capital = initial_capital
    trades = []
    
    for signal_type, date, price in signals:
        if signal_type == '买入':
            shares = capital / price
            capital = 0
        elif signal_type == '卖出':
            capital = shares * price
            shares = 0
        
        trades.append({
            'type': signal_type,
            'date': date,
            'price': price,
            'capital': capital,
            'shares': shares
        })
    
    # 最终收益
    final_value = capital + shares * nav_series.iloc[-1]
    total_return = (final_value - initial_capital) / initial_capital * 100
    
    # 计算胜率
    wins = 0
    total_trades = len(signals) // 2
    
    return {
        '初始资金': initial_capital,
        '最终价值': round(final_value, 2),
        '总收益率': round(total_return, 2),
        '交易次数': total_trades,
        '交易记录': trades
    }
```
